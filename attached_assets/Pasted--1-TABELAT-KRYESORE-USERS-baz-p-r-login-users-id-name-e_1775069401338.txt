✅ RREGULLI KRYESOR
🔥 çdo gjë filtrohet me teacher_id
👨‍🏫 Teacher sheh vetëm:
Courses
SELECT * FROM courses WHERE teacher_id = current_teacher_id
Students (vetëm të tijtë)
SELECT students.*
FROM students
JOIN course_student ON students.id = course_student.student_id
JOIN courses ON courses.id = course_student.course_id
WHERE courses.teacher_id = current_teacher_id
Lessons
SELECT lessons.*
FROM lessons
JOIN courses ON lessons.course_id = courses.id
WHERE courses.teacher_id = current_teacher_id
Quizzes
SELECT quizzes.*
FROM quizzes
JOIN courses ON quizzes.course_id = courses.id
WHERE courses.teacher_id = current_teacher_id
👨‍🎓 Student sheh vetëm:
Courses ku është i regjistruar
SELECT courses.*
FROM courses
JOIN course_student ON courses.id = course_student.course_id
WHERE course_student.student_id = current_student_id

🔥 BEST PRACTICE (Laravel)

Në Model:

Course.php
public function teacher()
{
    return $this->belongsTo(Teacher::class);
}
Teacher.php
public function courses()
{
    return $this->hasMany(Course::class);
}
Filter global (shumë powerful)
Course::where('teacher_id', auth()->user()->teacher->id)
